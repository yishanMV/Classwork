void defuseTwoBombs()
{
    moveUntilDetect();
    pickBeeper();
    turnAround();
    moveUntilDetect();
    turnRight();
    moveUntilDetect();
    pickBeeper();
    turnAround();
    moveUntilDetect();
    turnAround();
    
}

void moveUntilDetect()
{
    while(! onBeeper() && frontIsClear())
    {
        moveForward();
    }
}
