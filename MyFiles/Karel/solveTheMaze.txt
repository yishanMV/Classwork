void solveTheMaze()
{
    while(! onBeeper())
    {
        DeterminePath();
    }
}

void DeterminePath()
{
    if(leftIsClear())
    {
        turnLeft();
        moveForward();
    }
    else
    {
        turnRight();
    }
    
}
